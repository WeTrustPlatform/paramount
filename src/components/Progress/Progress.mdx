## Usage

### Default

<Playground>
  <Counter initial={0}>
    {({ count, incBy, decBy }) => (
      <Box>
        <Progress percent={count} />
        <Box flexDirection="row">
          <Button onPress={() => incBy(10)} title="Inc 10%" />
          <Button onPress={() => decBy(10)} title="Dec 10%" />
        </Box>
      </Box>
    )}
  </Counter>
</Playground>
