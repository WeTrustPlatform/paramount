## Usage

### Default

<Playground>
  <Counter initial={0}>
    {({ count, incBy, decBy }) => (
      <Box>
        <Progress percent={count} />
        <Box flexDirection="row">
          <Button onPress={() => incBy(10)}>Inc 10%</Button>
          <Button onPress={() => decBy(10)}>Dec 10%</Button>
        </Box>
      </Box>
    )}
  </Counter>
</Playground>
