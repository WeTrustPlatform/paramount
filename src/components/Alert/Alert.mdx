# Alert

Give feedback to the user about an action or state

## Usage

<Playground>
  <Box paddingVertical={8}>
    <Alert title="Info" description="Message" />
  </Box>
  <Box paddingVertical={8}>
    <Alert intent="success" title="Success" description="Message" />
  </Box>
  <Box paddingVertical={8}>
    <Alert intent="warning" title="Warning" description="Message" />
  </Box>
  <Box paddingVertical={8}>
    <Alert
      intent="danger"
      title="Danger"
      description="Message"
      overrides={{
        Action: {
          component: () => (
            <Button title="Action" appearance="minimal" color="primary" />
          ),
        },
      }}
    />
  </Box>
</Playground>

<Props of={Alert} />

## Customization

```tsx
<Root>
  <LeftWrapper>
    <AlertIcon />
    <Body>
      <Title />
      <Description />
    </Body>
  </LeftWrapper>
  <Action />
</Root>
```
