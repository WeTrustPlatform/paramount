### Controlled

<Playground>
  <State initial={{ activeTabIndex: 0 }}>
    {({ state, setState }) => (
      <Tabs
        distribution="scrollable"
        activeTabIndex={state.activeTabIndex}
        onPress={index => setState({ activeTabIndex: index })}
      >
        <Tab
          icon={
            <Icon
              name="home"
              size={24}
              color={state.activeTabIndex === 0 ? '#1b877a' : '#7e7e7e'}
            />
          }
        />
        <Tab title="Tab 2" />
        <Tab isDisabled title="Tab 3" />
      </Tabs>
    )}
  </State>
</Playground>
