# ProgressBar

## Usage

<Playground>
  <State initial={{ value: 0 }}>
    {({ state, setState }) => (
      <Box>
        <ProgressBar percent={state.value} size="large" />
        <Counter
          value={state.value}
          onValueChange={value => setState({ value })}
          min={0}
          max={100}
          step={10}
        />
      </Box>
    )}
  </State>
</Playground>
