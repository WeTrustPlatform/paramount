import { Counter as CounterContainer } from 'react-powerplug';

# ProgressBar

### Usage

<Playground>
  <CounterContainer initial={0}>
    {({ count, incBy, decBy }) => (
      <Box>
        <ProgressBar
          percent={count}
          size="large"
          getStyles={(props, theme) => ({
            containerStyle: {},
            progressStyle: {},
          })}
        />
        <Counter
          count={count}
          onDecrement={() => decBy(10)}
          onIncrement={() => incBy(10)}
          min={0}
          max={100}
        />
      </Box>
    )}
  </CounterContainer>
</Playground>
