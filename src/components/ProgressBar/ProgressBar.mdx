import { Playground, Props } from 'docz';

import { Box, Counter, CounterState, ProgressBar } from '..';

# ProgressBar

### Usage

<Playground>
  <CounterState initial={0}>
    {({ count, incrementBy, decrementBy }) => (
      <Box>
        <ProgressBar
          percent={count}
          size="large"
          getStyles={(props, theme) => ({
            containerStyle: {},
            progressStyle: {},
          })}
        />
        <Counter
          count={count}
          onDecrement={() => decrementBy(10)}
          onIncrement={() => incrementBy(10)}
          min={0}
          max={100}
        />
      </Box>
    )}
  </CounterState>
</Playground>
