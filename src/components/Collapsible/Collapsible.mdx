import { Playground, Props } from 'docz';
import { Box, Collapsible, Toggle } from '..';

# Collapsible

Component to reveal and hide content

### Usage

<Playground>
  <Toggle initial={false}>
    {({ on, toggle }) => (
      <Collapsible
        title={on ? 'Close collapsible' : 'Open collapsible'}
        isOpen={on}
        onOpen={toggle}
        onClose={toggle}
      >
        <Box
          marginVertical={16}
          justifyContent="center"
          alignItems="center"
          height={300}
          backgroundColor="#67c6bb"
        />
      </Collapsible>
    )}
  </Toggle>
</Playground>

### Props

<Props of={Collapsible} />

### Customization

Using `getStyles` prop

```
CollapsibleStyles {
  touchableStyle: ViewStyle;
  outerWrapperStyle: ViewStyle;
  contentWrapperStyle: ViewStyle;
  textStyle: ViewStyle;
  iconWrapperStyle: ViewStyle;
}

getStyles={(CollapsibleProps, Theme) => CollapsibleStyles}
```

Markup

```tsx
<>
  <TouchableOpacity touchableStyle>
    <View outerWrapperStyle>
      <Text textStyle>{title}</Text>
      <View iconWrapperStyle>{icon}</View>
    </View>
  </TouchableOpacity>
  <View contentWrapperStyle>{children}</View>
</>
```
