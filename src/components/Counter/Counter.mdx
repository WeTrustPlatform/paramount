## Usage

### Counter

<Playground>
  <CounterContainer initial={1}>
    {({ count, inc, dec }) => (
      <Counter count={count} onIncrement={inc} onDecrement={dec} />
    )}
  </CounterContainer>
</Playground>

### Counter with min and max count

<Playground>
  <CounterContainer initial={0}>
    {({ count, inc, dec }) => (
      <Counter
        count={count}
        onDecrement={dec}
        onIncrement={inc}
        min={0}
        max={10}
      />
    )}
  </CounterContainer>
</Playground>
