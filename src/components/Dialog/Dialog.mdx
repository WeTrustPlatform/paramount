import { Playground, Props } from 'docz';

import { Box, Button, Dialog, Heading, Icon, Text, Toggle } from '..';

# Dialog

Display centered window on the screen over a dark overlay

## Usage

<Playground>
  <Toggle initial={false}>
    {({ on, toggle }) => (
      <Box>
        <Dialog
          isVisible={on}
          onRequestClose={toggle}
          overrides={{
            Header: {
              component: () => <Heading>Header</Heading>,
            },
            Footer: {
              component: () => <Heading>Footer</Heading>,
            },
          }}
        >
          <Box height={1800}>
            <Text>Put any content in the dialog</Text>
          </Box>
        </Dialog>
        <Button onPress={toggle} title="Open dialog" />
      </Box>
    )}
  </Toggle>
</Playground>
