import { Playground, PropsTable } from 'docz';
import Badge from './Badge';
import { ThemeConsumer } from '../../theme';
import { Box } from '../Layout';

## Usage

**Note** Accessibility/Interactivity is a responsibility of a wrapping component

### Default

<Playground>
  <ThemeConsumer>
    {theme => (
      <Box>
        {Object.keys(theme.themeVariables.fills.solid).map(color => (
          <Box flexDirection="row" key={color}>
            <Badge color={color}>{color}</Badge>
            <Badge color={color} isSolid>
              {color}
            </Badge>
          </Box>
        ))}
      </Box>
    )}
  </ThemeConsumer>
</Playground>
