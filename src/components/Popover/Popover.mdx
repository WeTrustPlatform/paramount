### Popover from a target node

<Spacing paddingY={10}>
  <Toggle>
    {({ on, toggle }) => (
      <RefMeasure>
        {({ forwardRef, onLayout, measurements }) => {
          return (
            <>
              <Popover
                targetMeasurements={measurements}
                isVisible={on}
                onClose={toggle}
                position="top-right"
                content={
                  <Box backgroundColor="white">
                    <Text>
                      Long text that should be well visible and non-obstructing
                      Long text that should be well visible and non-obstructing
                    </Text>
                  </Box>
                }
              />
              <Button
                onLayout={onLayout}
                ref={forwardRef}
                onPress={toggle}
                isInline
              >
                Some Button
              </Button>
            </>
          );
        }}
      </RefMeasure>
    )}
  </Toggle>
</Spacing>
