import { Playground, PropsTable } from 'docz';
import { Popover } from './Popover';
import { Button } from '../Button';
import { Box, Spacing } from '../Layout';
import { Text } from '../Typography';
import { Toggle } from 'react-powerplug';

export default ({ children }) => {
  document.body.style.height = 'initial';
  document.body.style.position = 'relative';
  return <div>{children}</div>;
};

## Usage

`Popover` composes `Positioner` and adds only a white box wrapping. Content that can be anything. Popover just applies some styles for the content. For more low-level usage, use `Positioner`

### Popovers with fixed width and height content
