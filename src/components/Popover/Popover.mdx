# Popover

Composes `Positioner`

### Usage

<Playground>
  <Box height={250}>
    <Toggle>
      {({ on, toggle }) => (
        <Popover
          isVisible={on}
          position="right"
          getStyles={(props, theme) => ({
            positionerStyle: {},
            modalContainerStyle: {},
            popoverStyle: {},
          })}
          content={
            <Box width={200} height={200}>
              <Text>
                Due to variety and complexity involved in detecting touch
                outside of the component on the web and in native, you have to
                manage its open and closed state
              </Text>
              <Button
                appearance="outline"
                color="danger"
                onPress={toggle}
                title="Close popover"
              />
            </Box>
          }
        >
          <Button onPress={toggle} title="Open popover" />
        </Popover>
      )}
    </Toggle>
  </Box>
</Playground>
