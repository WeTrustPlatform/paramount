# TextInput

Composes React Native's `TextInput`

## Usage

<Playground>
  <State initial={{ value: '' }}>
    {({ state, setState }) => (
      <TextInput
        value={state.value}
        placeholder="Placeholder"
        size="large"
        onValueChange={value => setState({ value })}
        isInvalid={false}
        isClearable
        onClear={() => console.log('Cleared!')} // Only triggered with `isClearable`
      />
    )}
  </State>
</Playground>
