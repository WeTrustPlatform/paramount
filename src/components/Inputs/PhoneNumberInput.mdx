import { Playground, Props } from 'docz';
import { State } from 'react-powerplug';

import { Heading } from '../Typography';
import { Box } from '../Box';

import { PhoneNumberInput } from './PhoneNumberInput';

# PhoneNumberInput

Appends a button to select country code

### Usage

<Playground>
  <State initial={{ countryCode: 'US', phoneNumber: '' }}>
    {({ state, setState }) => (
      <PhoneNumberInput
        header={
          <Box marginVertical={24} paddingHorizontal={16}>
            <Heading size="xxxlarge">Select your country</Heading>
          </Box>
        }
        onChangeCountryCode={countryCode => setState({ countryCode })}
        onChangePhoneNumber={phoneNumber => setState({ phoneNumber })}
        phoneNumber={state.phoneNumber}
        countryCode={state.countryCode}
        placeholder="Enter your phone number"
      />
    )}
  </State>
</Playground>

### Props

<Props of={PhoneNumberInput} />
