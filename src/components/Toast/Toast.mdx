import { Playground, PropsTable } from 'docz';
import { ToastContext, ToastProvider } from '.';
import { Button } from '../Button';
import { Box } from '../Layout';
import Toast from './Toast';

## Usage

Composes `Alert`

<Playground>
  <ToastProvider>
    <ToastContext.Consumer>
      {({ notify }) => (
        <>
          <Button
            onPress={() =>
              notify({
                title: 'Title',
                description: 'Description',
                isCloseable: true,
              })
            }
            title="Open default toast"
          />
          <Button
            onPress={() =>
              notify({
                description: 'Description',
                intent: 'success',
                title: 'Title',
              })
            }
            title="Open success toast"
          />
          <Button
            onPress={() =>
              notify({
                description: 'Description',
                intent: 'danger',
                title: 'Title',
              })
            }
            title="Open danger toast"
          />
          <Button
            onPress={() =>
              notify({
                description: 'Description',
                intent: 'warning',
                title: 'Title',
              })
            }
            title="Open warning toast"
          />
        </>
      )}
    </ToastContext.Consumer>
  </ToastProvider>
</Playground>
