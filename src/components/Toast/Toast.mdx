# Toast

Composes `Alert`.

### Usage

<Playground>
  <ToastProvider
    getStyles={(props, theme) => ({
      containerStyle: {},
    })}
  >
    <ToastContext.Consumer>
      {({ notify }) => (
        <Button
          onPress={() => {
            notify({
              title: 'Title',
              description: 'Description',
              component: null, // A React Component that overrides title and description
              isCloseable: true,
              offset: 16,
              duration: 3000,
              // ...Alert props e.g. getStyles
            });
          }}
          title="Use"
        />
      )}
    </ToastContext.Consumer>
  </ToastProvider>
</Playground>
