import { Box, Visible, LayoutProvider, Text } from '..';

# Visible

Renders the children if the current screen size matches any of defined size prop

### Usage

<Playground>
  <LayoutProvider>
    <Visible large xlarge>
      <Box backgroundColor="#67c6bb" height={48}>
        <Text align="center" color="white">
          Large and greater
        </Text>
      </Box>
    </Visible>
    <Visible medium>
      <Box backgroundColor="#1ca5F7" height={48}>
        <Text align="center" color="white">
          Medium
        </Text>
      </Box>
    </Visible>
    <Visible xsmall small>
      <Box backgroundColor="#d26557" height={48}>
        <Text align="center" color="white">
          Small and lower
        </Text>
      </Box>
    </Visible>
  </LayoutProvider>
</Playground>
